import random
#import math
import operator
import matplotlib.pyplot

# Creating an empty list.

agents = []


# Adding two random integer numbers between 0 and 99 to the list and printing
# it.

agents.append([random.randint (0, 99), random.randint (0, 99)])


print(agents)


agents.append([random.randint (0, 99), random.randint (0, 99)])


print(agents)

# Finding the max value between the two y coordinates.
#print (max(agents))


# Finding the max value in the agents list, however, only looking at
# values that are in second elements i.e. 1 i.e. x values.

print(max(agents, key=operator.itemgetter(1)))


# Setting the y and x limits of the plot axes between 0 and 99.

matplotlib.pyplot.ylim(0, 99)


matplotlib.pyplot.xlim(0, 99)


# Creating a scatter plot of x vs y. e.g. for [[26, 43], [12, 43]]
# it would be 43 and 26. This is the point further east and the
# colour is changed to 'red'.

matplotlib.pyplot.scatter(agents[0][1], agents[0][0], color= 'red')


# Same as above but for the other combination of y and xs.

matplotlib.pyplot.scatter(agents[1][1], agents[1][0])


# Displaying the scatter plot.

matplotlib.pyplot.show()
